    Append `POST` to `app/api/goals/route.ts` (same file, after `GET`):
    ```typescript
    export async function POST(request: Request) {
      try {
        const userId = await getOrCreateSystemUserId()
        const body = await request.json()

        const targetWeight = typeof body.targetWeight === 'number' && Number.isFinite(body.targetWeight) ? body.targetWeight : null
        const targetBodyFat = typeof body.targetBodyFat === 'number' && Number.isFinite(body.targetBodyFat) ? body.targetBodyFat : null
        const targetDate = typeof body.targetDate === 'string' && body.targetDate ? new Date(body.targetDate) : null

        if (targetWeight === null && targetBodyFat === null) {
          return NextResponse.json(
            { error: 'Au moins un objectif chiffré (poids cible ou % masse grasse cible) est requis' },
            { status: 400 }
          )
        }

        const existing = await getCurrentGoal(userId)
        const goal = await upsertGoal(userId, { targetWeight, targetBodyFat, targetDate })
        return NextResponse.json({ goal }, { status: existing ? 200 : 201 })
      } catch (error) {
        console.error('[API] Error saving goal:', error)
        return NextResponse.json(
          { error: 'Impossible d\'enregistrer l\'objectif', details: error instanceof Error ? error.message : 'Unknown error' },
          { status: 500 }
        )
      }
    }
    ```
    Note: `existing` is fetched before `upsertGoal` purely to determine the correct response status code (201 create vs 200 replace) for the client — `upsertGoal` internally does its own `getCurrentGoal` lookup too (from Plan 01), so this is a second cheap read, not a race condition risk in this single-user app.
