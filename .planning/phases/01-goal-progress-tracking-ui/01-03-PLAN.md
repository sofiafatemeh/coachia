      title, points, target, unit,
    }: {
      title: string
      points: { date: string; raw: number | null; smoothed: number | null }[]
      target: number | null
      unit: string
    }) {

        return (

          <ResponsiveContainer width="100%" height={280}>
            <LineChart data={points}>
              <CartesianGrid stroke="var(--gold-soft)" strokeDasharray="3 3" />
              <XAxis dataKey="date" tick={{ fontSize: 12 }} />

              <Tooltip />
              <Legend />
